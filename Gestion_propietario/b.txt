CREATE VIEW [dbo].[cuenta_ver]
AS
SELECT        dbo.cuenta.id_cuenta, dbo.concepto.concepto, dbo.cuenta.fe_desde, dbo.cuenta.fe_hasta, dbo.cuenta.monto, dbo.estado_cuenta.estado
FROM            dbo.cuenta INNER JOIN
                         dbo.concepto ON dbo.cuenta.id_concepto = dbo.concepto.id_concepto INNER JOIN
                         dbo.estado_cuenta ON dbo.cuenta.id_estado_cuenta = dbo.estado_cuenta.id_estado_cuenta
