SELECT
    gestion_de_propietario.villas.id_villa AS id_villa,
        gestion_de_propietario.grupo_villa.grupo,
        gestion_de_propietario.villas.numero,
    gestion_de_propietario.villas.cont_eeh AS cont_eeh,
    gestion_de_propietario.villas.habitaciones AS habitaciones,
    gestion_de_propietario.villas.lote AS lote,
    gestion_de_propietario.condicion_villas.condicion AS condicion,
    gestion_de_propietario.villas.id_estado AS id_estado,
    gestion_de_propietario.estado_villa.estado AS estado,
    gestion_de_propietario.villas.modelo AS modelo
FROM
    (
        (
            (
                gestion_de_propietario.villas
            JOIN gestion_de_propietario.grupo_villa ON
                (
                    (
                        gestion_de_propietario.villas.id_grupo = gestion_de_propietario.grupo_villa.id_grupo
                    )
                )
            )
        JOIN gestion_de_propietario.condicion_villas ON
            (
                (
                    gestion_de_propietario.villas.id_condicion = gestion_de_propietario.condicion_villas.id_condicion
                )
            )
        )
    JOIN gestion_de_propietario.estado_villa ON
        (
            (
                gestion_de_propietario.villas.id_estado = gestion_de_propietario.estado_villa.id_estado
            )
        )
    )
